program a {
	vars int k;

	func fibo(int i) returns int {
		vars int val;

		if (i < 2) {
			val = i;
		} else {
			val = fibo(i - 1) + (fibo(i - 2));
		}

		return val;
	}

	func factorial(int n) returns int {
		vars int val;

		if (n <= 1) {
			val = 1;
		} else {
			val = n * (factorial(n - 1));
		}

		return val;
	}

	main {
		vars int i;
		k = 0;
		i = 3;

		print("Fibo: ", fibo(i));
		print("Fact: ", factorial(factorial(i)));
	}
}
